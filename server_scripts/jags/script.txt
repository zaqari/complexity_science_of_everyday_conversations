# Inferring the Mean and Standard Deviation of a Gaussian
model{
  # General params
  nxbeta ~ dunif(0,1000)
  nybeta ~ dunif(-1,1)

  # estimating diagonal variances
  for (k in 1:K){
    sigma_[k] ~ dunif(.01,100)
  }

  # mechanism for calcualting beta weights for covariance
  #  while maintaining definite positive status . . .
  #  I think this code here is pretty hoaky.
  for (c1 in 1:K){
    for (c2 in 1:K){
      maxima[c1,c2] <- min(sigma_[c1]^2, 1)
      v[c1,c2] ~ dunif(0,maxima[c1,c2]/(K-1))
      r[c1,c2] <- ifelse(c1<c2,v[c1,c2],0) + ifelse(c1==c2,1,0) + ifelse(c1>c2,0,0)
    }
  }

  # creating covariance matrix
  for (k in 1:K){
    for (l in 1:K){
       Sigma[k,l] <- ifelse(k<=l,r[k,l],r[l,k]) * sigma_[k] * sigma_[l]
    }
  }

  for (k in 1:(K-1)){
    mu[k] ~ dnorm(gamma[k], phi[k])
  }
  mu[K] ~ dunif(0,.0001)

  # creating location params and sigma
  for (k in 1:K){
    gamma[k] ~ dunif(-1000, 1000)
    phi[k] ~ dunif(.0001, 100)
  }

  # Log-Likelihood Constants
  S <- exp(logdet(Sigma))
  PI_ <- pow(2*PI, -K/2)
  CONST <- S * PI_

  # group variance
  for (g in 1:GROUPS){
    lambda[g] ~ dunif(1,10)
    # theta[g] ~ dunif(1,10)
  }

  # results
  for (i in 1:ROWS){

    # Custom Log-Likelihood Function
    e[i,1:K] <- x[i,] - mu

    for (ki in 1:K){
        first_in_prod[i,ki] <- inprod(e[i,], Sigma[,ki])
    }

    sinprod[i] <- inprod( first_in_prod[i,],  e[i,])
    L[i] <- CONST * exp(-.5 * sinprod[i])

    # Contribution due to average entropy
    BETA[i] <- (nxbeta * nx[i]) + (nybeta * ny[i])

    # estimate entropy value
    H[i] ~ dnorm(BETA[i] - log(L[i] + .000001), lambda[group[i]])
  }
}
